Пример 6.3. headerfy
#!/usr/bin/perl
# headerfy: оформление заголовков глав в HTML
$/ = '';
while ( <> ) {              # Получить абзац
    s{
        \A                  # Начало записи
        (                   # Сохранить в $1
            Chapter         # Текстовая строка
            \s+             # Обязательный пропуск
            \d+             # Десятичное число
            \s*             # Необязательный пропуск
            :               # Двоеточие
            . *             # Все, кроме перевода строки, до конца строки
        )
    }{<H1>$1</H1>}gx;
    print;
}
