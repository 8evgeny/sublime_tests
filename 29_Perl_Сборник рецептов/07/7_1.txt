#!/usr/bin/perl
# splitwulog - разбиение журнала wuftpd по пользовател€м
use FileCache;
$outdir = "/var/log/ftp/by-user";
while (<>) {
    unless (defined ($user = (split)[-4])) {
       warn "Invalid line: $.\n";
       next;
    }
    $path = "$outdir/$user";
    cacheout $path;
    print $path $_;

}
