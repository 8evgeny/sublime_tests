Пример 16.9. dateplan
#!/usr/bin/perl -w
# dateplan - вывод текущей даты и времени в файл .plan
while (1) {
    open(FIFO, "> $ENV{HOME}/.plan")
        or die "Couldn’t open $ENV{HOME}/.plan for writing: $!\n";
    print FIFO "The current time is ", scalar(localtime), "\n";
    close FIFO;
    sleep 1;
}
