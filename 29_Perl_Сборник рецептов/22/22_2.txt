Пример 22.2. dom-titledumper
#!/usr/bin/perl -w
# dom-titledumper -- вывод элементов title iз файла books
#                    с использованием DOM

use XML::LibXML;
use Data::Dumper;
use strict;

my $parser = XML::LibXML->new;
my $dom = $parser->parse_file("books.xml") or die;

# Получить все элементы title
my @titles = $dom->getElementsByTagName("title");
foreach my $t (@titles) {
    # Получить текстовый узел внутри <title>
    # и вывести его значение.
    print $t->firstChild->data, "\n";
}
