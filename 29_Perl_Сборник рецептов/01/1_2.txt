Пример 1.2. randcap
#!/usr/bin/perl –p
# randcap: фильтр, который случайным образом 
# преобразует к верхнему регистру 20% символов
# Начиная с версии 5.4, вызов srand() необязателен.

BEGIN { srand(time() ^ ($$ + ($$ << 15))) }
sub randcase { rand(100) < 20 ? "\u$_[0]" : "\l$_[0]" }
s/(\w)/randcase($1)/ge;
